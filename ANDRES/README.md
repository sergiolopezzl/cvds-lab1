## Mi primer codigo
```
num = int(input("Digite un numero:"))
divisor = 1
divisores = []
t_divisores = 0
num % divisor

while divisor <= num:
    if num % divisor == 0:
        divisores.append(divisor)
        divisor = divisor + 1
        t_divisores = t_divisores + 1
    elif num % divisor != 0:
        divisor = divisor + 1
        num % divisor

divisores.reverse()
print(divisores[0:])
print(f"{t_divisores} divisores")
```
